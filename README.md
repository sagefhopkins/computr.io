# computr.io
Computr.io
# Required packages
Flask
flask_mysqldb
wtforms
passlib.has
functools
os
io
base64
pillow
pgeocode

# Easy setup

    $ python3 -m venv computr.env
    $ source computr.env/bin/activate
    $ pip install -r requirements.txt

# To update requirements

    $ source computr.env/bin/activate
    $ pip install foo
    $ pip freeze > requirements.txt
